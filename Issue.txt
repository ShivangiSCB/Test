package com.sc.scroe.shared.util;

import org.springframework.dao.DataIntegrityViolationException;
import org.springframework.stereotype.Component;

@Component
public class DataIntegrityViolationParser {

    public boolean isDuplicateViolation(DataIntegrityViolationException ex, String tableName, String... columnNames) {
        Throwable rootCause = ex.getMostSpecificCause();
        String message = rootCause.getMessage();
        if (message == null) {
            message = ex.getMessage();
        }
        if (message == null) {
            return false;
        }

        String normalizedMessage = message.toLowerCase();
        if (!normalizedMessage.contains("duplicate key")) {
            return false;
        }

        if (tableName != null && !tableName.isBlank() && !normalizedMessage.contains(tableName.toLowerCase())) {
            return false;
        }

        if (columnNames == null) {
            return true;
        }

        for (String columnName : columnNames) {
            if (columnName != null && !columnName.isBlank() && !normalizedMessage.contains(columnName.toLowerCase())) {
                return false;
            }
        }

        return true;
    }
}
