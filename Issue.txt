    private List<String> splitCommaSeparatedValues(String csvValues) {
        if (csvValues == null || csvValues.isBlank()) {
            return Collections.emptyList();
        }

        String[] values = csvValues.split(",");
        List<String> items = new ArrayList<>();
        for (String value : values) {
            String trimmed = value.trim();
            if (!trimmed.isEmpty()) {
                items.add(trimmed);
            }
        }
        return items;
    }

    private List<Map<String, String>> buildMiscInfos(String infoCodes, String infoValues) {
        List<String> parsedCodes = splitCommaSeparatedValues(infoCodes);
        List<String> parsedValues = splitCommaSeparatedValues(infoValues);
        if (parsedCodes.isEmpty() && parsedValues.isEmpty()) {
            return Collections.emptyList();
        }

        int size = Math.max(parsedCodes.size(), parsedValues.size());
        List<Map<String, String>> miscInfos = new ArrayList<>();

        for (int i = 0; i < size; i++) {
            int sequence = i + 1;
            Map<String, String> miscInfo = new LinkedHashMap<>();
            miscInfo.put("infoCode" + sequence, i < parsedCodes.size() ? parsedCodes.get(i) : "");
            miscInfo.put("infoValue" + sequence, i < parsedValues.size() ? parsedValues.get(i) : "");
            miscInfos.add(miscInfo);
        }
        return miscInfos;
    }

    private boolean hasText(String value) {
        return value != null && !value.isBlank();
    }
}

@Component
public class SubledgerMasterAccountUtil {

    public void mapPostRequestToEntity(CreateMasterAccountRequest request, SubledgerMasteraccountEntity entity) {

        entity.setDealId(request.getDealId());
        entity.setClientId(request.getClientId());

        entity.setAccountNumber(request.getMasterAccountNumber());
        entity.setCountryCode(request.getMasterAccountCountry());
        entity.setCurrencyCode(request.getMasterAccountCurrencyCode());

        entity.setAccountName(request.getMasterAccountName());
        entity.setCompanyInitial(request.getCompanyInitial());

        entity.setClientType(request.getClientType());

        entity.setAccountType(request.getSubAccountType());

        entity.setSubProductCode(request.getProductCode());

        entity.setClassificationCode(request.getAccountClassificationCode());
        entity.setProductCode(request.getProductCode());

        entity.setSubliteKycCddFlag(request.getSubKycCddFlag());

        entity.setSubliteKycCddRiskCode(request.getSubKycCddRiskCode());

        entity.setSubDefaultRiskCodeFlag(request.getSubDefaultRiskCodeHoldFlag());

        if (request.getSubDefaultRiskCode() != null &&
                !request.getSubDefaultRiskCode().isEmpty()) {

            entity.setSubDefaultRiskCode(
                    String.join(",", request.getSubDefaultRiskCode()));
        }

        entity.setAddressInheritFlag(
                request.getMasterAccountAddressInheritFlag());

        entity.setClientsMoney(request.getClientsMoney());

        entity.setClientsMoneyValue(
                request.getInfoCodeClientsMoney());

        entity.setSubAccountClosureOffset(
                request.getSubAccountClosureOffset());

        if (request.getMiscInfos() != null) {

            request.getMiscInfos().forEach(miscInfoDTO -> {
                if (miscInfoDTO.getInfoCode1() != null) {

                    entity.setInfoCode1(miscInfoDTO.getInfoCode1());
                    entity.setInfoValue1(miscInfoDTO.getInfoValue1());
                }
                if (miscInfoDTO.getInfoCode2() != null) {

                    entity.setInfoCode2(miscInfoDTO.getInfoCode2());
                    entity.setInfoValue2(miscInfoDTO.getInfoValue2());
                }
            });
        }

        entity.setAccountBranch(request.getAccountBranch());

        entity.setConsolidatedStatementFlag(
                request.getConsolidatedStatementFlag());

        entity.setGlid(request.getGlid());

        entity.setMasterNumber(request.getMasterNumber());

        entity.setAccountCategory(request.getAccountCategory());

        entity.setDsrClosingId(request.getDsrClosingId());

        entity.setAdviseInterestIndicator(
                request.getAdviseInterestIndicator());

        entity.setIbanNumber(request.getIbanNumber());

        entity.setAccountClosedDate(request.getCloseDate());

        entity.setSubProductCode(request.getSubProductCode());

        entity.setCreditInterestCode(request.getCreditInterestCode());

        entity.setDebitInterestCode(request.getDebitInterestCode());

        entity.setInterestCode(request.getInterestCode());

        entity.setSuppressStatementFlag(
                request.getSuppressStatementFlag());

        entity.setCreditSubAcctIntToMast(
                request.getCreditSubAccountInterestToMasterAccount());

        entity.setMaxSubAccountLimit(
                request.getMaxSubAccountLimit());

        entity.setAccountOpenedDate(
                request.getOpenedOn());

        entity.setAccountStatus(
                request.getEbbsAccountStatus());
    }

import org.springframework.dao.DataIntegrityViolationException;
import org.springframework.stereotype.Component;

@Component
public class DataIntegrityViolationParser {

    public boolean isDuplicateViolation(DataIntegrityViolationException ex, String tableName, String... columnNames) {
        Throwable rootCause = ex.getMostSpecificCause();
        String message = rootCause.getMessage();
        if (message == null) {
            message = ex.getMessage();
        }
        if (message == null) {
            return false;
        }

        String normalizedMessage = message.toLowerCase();
        if (!normalizedMessage.contains("duplicate key")) {
            return false;
        }

        if (tableName != null && !tableName.isBlank() && !normalizedMessage.contains(tableName.toLowerCase())) {
            return false;
        }

        if (columnNames == null) {
            return true;
        }

        for (String columnName : columnNames) {
            if (columnName != null && !columnName.isBlank() && !normalizedMessage.contains(columnName.toLowerCase())) {
                return false;
            }
        }

        return true;
    }
