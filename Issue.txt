package com.sc.scroe.controller;

import jakarta.validation.Valid;

import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

@RestController
@RequestMapping("/v1/subledger/patterns")
public class PatternController {

    private static final Logger logger = LoggerFactory.getLogger(PatternController.class);

    @Autowired
    PatternService patternService;

    @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
    public ResponseEntity<GetPatternsResponse> getPattern(
            @RequestParam(value = "clientId", required = false) String clientId,
            @RequestParam(value = "dealId", required = false) String dealId,
            @RequestParam(value = "patternType", required = false) String patternType,
            @RequestParam(value = "page", defaultValue = "1") Integer page,
            @RequestParam(value = "pageSize", defaultValue = "10") Integer pageSize) {
        GetPatternsResponse response = patternService.getPattern(clientId, dealId, patternType, page,
                pageSize);
        return NoRecordsResponseUtil.returnResponseOrThrowIfEmpty(
                response,
                response.getData(),
                "No subledger pattern records found",
                NoRecordsResponseUtil.formatCriterion("clientId", clientId),
                NoRecordsResponseUtil.formatCriterion("dealId", dealId),
                NoRecordsResponseUtil.formatCriterion("patternType", patternType));
    }

    @PostMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
    public ResponseEntity<CreatePatternResponse> createPattern(
            @Valid @RequestBody CreatePatternRequest request) {
        Long patternId = patternService.createPattern(request);
        logger.info("Created subledger pattern with patternId={}", patternId);
        return new  ResponseEntity<>(new CreatePatternResponse(patternId),HttpStatus.CREATED);
    }
}

@RestControllerAdvice
public class GlobalExceptionHandler {

    private static final Logger logger = LoggerFactory.getLogger(GlobalExceptionHandler.class);

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ErrorResponse> handleValidation(MethodArgumentNotValidException ex) {
        String details = ex.getBindingResult()
                .getFieldErrors()
                .stream()
                .map(err -> err.getField() + ": " + err.getDefaultMessage())
                .reduce((a, b) -> a + "; " + b)
                .orElse(ex.getMessage());
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        Map<String, String> error = new LinkedHashMap<>();
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", details);
        return ResponseEntity
                .status(HttpStatus.BAD_REQUEST)
                .body(new ErrorResponse(Arrays.asList(error)));
    }

    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<ErrorResponse> handleBadRequest(IllegalArgumentException ex) {
        Map<String, String> error = new LinkedHashMap<>();
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", ex.getMessage());
        return ResponseEntity
                .status(HttpStatus.BAD_REQUEST)
                .body(new ErrorResponse(Arrays.asList(error)));
    }

    @ExceptionHandler(ResourceNotFoundException.class)
    public ResponseEntity<ErrorResponse> handleResourceNotFound(ResourceNotFoundException ex) {
        Map<String, String> error = new LinkedHashMap<>();
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", ex.getMessage());
        return ResponseEntity
