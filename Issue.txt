    @Override
    public SubledgerPatternResponse getPattern(String clientId, String dealId, String patternType, Integer page, Integer pageSize) {
        logger.info("Retrieving subledger patterns for clientId={}, dealId={}, patternType={}, page={}, pageSize={}", clientId, dealId, patternType, page, pageSize);
        Pageable pageable = patternPageService.createPageable(page, pageSize);

        Page<SubledgerPatternEntity> pageResult;
        if (patternType != null && !patternType.isBlank()) {
            pageResult = subledgerPatternRepository.findAllByClientIdAndDealIdAndPatternTypeAndDeleteFlagFalse(
                    clientId,
                    dealId,
                    patternType.trim().toUpperCase(),
                    pageable);
        } else {
            pageResult = subledgerPatternRepository.findAllByClientIdAndDealIdAndDeleteFlagFalse(clientId, dealId, pageable);
        }

        return buildResponse(pageResult);
    }
