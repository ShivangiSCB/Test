package com.sc.scroe.pattern.specification;

import jakarta.persistence.criteria.Predicate;
import org.springframework.data.jpa.domain.Specification;

import com.sc.scroe.pattern.entity.SubledgerPatternEntity;

import java.util.ArrayList;
import java.util.List;

public class SubledgerPatternSpecification {

    public static Specification<SubledgerPatternEntity> filterBy(
            String clientId,
            String dealId,
            String patternType) {

        return (root, query, cb) -> {

            List<Predicate> predicates = new ArrayList<>();

            // Always fetch only non-deleted records
            predicates.add(cb.isFalse(root.get("deleteFlag")));

            if (clientId != null && !clientId.isBlank()) {
                predicates.add(cb.equal(root.get("clientId"), clientId.trim()));
            }

            if (dealId != null && !dealId.isBlank()) {
                predicates.add(cb.equal(root.get("dealId"), dealId.trim()));
            }

            if (patternType != null && !patternType.isBlank()) {
                predicates.add(cb.equal(
                        cb.upper(root.get("patternType")),
                        patternType.trim().toUpperCase()));
            }

            return cb.and(predicates.toArray(new Predicate[0]));
        };
    }
}
package com.sc.scroe.pattern.util;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.data.domain.Page;

import com.sc.scroe.pattern.constant.PatternConstants;
import com.sc.scroe.pattern.entity.SubledgerPatternEntity;
import com.sc.scroe.pattern.repository.SubledgerPatternRepository;
import com.sc.scroe.pattern.request.SubledgerPatternPostRequest;
import com.sc.scroe.pattern.response.SubledgerPatternResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class SubledgerPatternUtil {

    private static final Logger logger = LoggerFactory.getLogger(SubledgerPatternUtil.class);


    @Autowired
    SubledgerPatternRepository subledgerPatternRepository;

     public SubledgerPatternResponse buildResponse(Page<SubledgerPatternEntity> pageResult) {
        SubledgerPatternResponse response = new SubledgerPatternResponse();
        response.setPage(pageResult.getNumber() + 1);
        response.setPageCount(pageResult.getTotalPages());
        response.setTotalCount(pageResult.getTotalElements());
        response.setData(pageResult.getContent());
        return response;
    }

    public void validatePatternType(String patternType) {
        if (!PatternConstants.PatternType.isValid(patternType)) {
            logger.warn("Invalid patternType received: {}", patternType);
            throw new IllegalArgumentException("patternType must be one of: SUBACCOUNT, VA, HYBRID");
        }
    }

    public void validatePatternfields(String clientId, String dealId, String patternType) {
        SubledgerPatternEntity findfirst = subledgerPatternRepository
                .findAllByClientIdAndDealIdAndPatternTypeAndDeleteFlagFalse(clientId, dealId, patternType, null)
                .stream().findFirst().orElse(null);
        if (findfirst != null) {
            logger.warn("Subledger pattern with clientId={}, dealId={}, patternType={} already exists", clientId,
                    dealId, patternType);
            throw new IllegalArgumentException(
                    "Subledger pattern with the same clientId, dealId, and patternType already exists.");
        }
    }

    public void mapRequestToEntity(SubledgerPatternPostRequest request, SubledgerPatternEntity entity) {
        entity.setClientId(request.getClientId());
        entity.setDealId(request.getDealId());
        entity.setPatternType(request.getPatternType().trim().toUpperCase());
    }

}
