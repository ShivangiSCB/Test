import org.springframework.data.domain.Page;
import org.springframework.data.domain.Pageable;
import org.springframework.data.jpa.domain.Specification;
import org.springframework.stereotype.Service;

import java.time.LocalDateTime;

@Service
public class MasterAccountServiceImpl implements MasterAccountService {

    @Autowired
    MasterAccountRepository masteraccountRepository;

    @Autowired
    MasterAccountUtil masterAccountUtil;

    @Autowired
    DataIntegrityViolationParser dataIntegrityViolationParser;

    @Autowired
    PatternUtil patternUtil;

    @Override
    public CreateMasterAccountResponse createMasterAccount(CreateMasterAccountRequest request) {
        MasterAccountEntity entity = new MasterAccountEntity();
        masterAccountUtil.mapPostRequestToEntity(request, entity);
        MasterAccountEntity masteraccountEntity;
        try {
            masteraccountEntity = masteraccountRepository.save(entity);
        } catch (DataIntegrityViolationException ex) {
            if (dataIntegrityViolationParser.isDuplicateViolation(ex, MasterAccountConstants.MASTER_TABLE_NAME, MasterAccountConstants.MASTER_UNIQUE_COLUMNS)) {
                throw new DuplicateResourceException(MasterAccountConstants.DUPLICATE_MASTER_MESSAGE);
            }
            throw ex;
        }
        return new CreateMasterAccountResponse(masteraccountEntity.getMasterNumber());
    }

    @Override
    public UpdateMasterAccountResponse updateMasterAccount(Long masterAccountId, UpdateMasterAccountRequest request) {

        MasterAccountEntity entity = masteraccountRepository.findById(masterAccountId).orElseThrow(() -> new ResourceNotFoundException("No master account setup record found for the given masteraccountid", ApiWebConstants.API_COMMON_ERROR_CODE));
        masterAccountUtil.mapPutRequestToEntity(request.getNewValues(), entity);
        entity.setUpdatedAt(LocalDateTime.now()); //Check and remove if not required

        try {
            masteraccountRepository.save(entity);
        } catch (DataIntegrityViolationException ex) {
            if (dataIntegrityViolationParser.isDuplicateViolation(ex, MasterAccountConstants.MASTER_TABLE_NAME, MasterAccountConstants.MASTER_UNIQUE_COLUMNS)) {
                throw new DuplicateResourceException(MasterAccountConstants.DUPLICATE_MASTER_MESSAGE);
            }
            throw ex;
        }
        return new UpdateMasterAccountResponse(entity.getDealId(), entity.getClientId(), entity.getAccountNumber(), entity.getCountryCode(), entity.getCurrencyCode(), "Master account setup updated successfully", entity.getMasterNumber());
    }

    @Override
    public GetMasterAccountResponse getMasterAccount(String dealId, String clientId, String masterAccountNumber, String masterAccountCountry, String masterAccountCurrencyCode, int page, int size) {
        Pageable pageable = patternUtil.createPageable(page, size);
        Specification<MasterAccountEntity> specification =
                MasterAccountSpecification.filterBy(clientId,
                        dealId, masterAccountNumber,
                        masterAccountCountry,
                        masterAccountCurrencyCode);
        Page<MasterAccountEntity> pageResult = masteraccountRepository.findAll(specification, pageable);
        return masterAccountUtil.buildResponse(pageResult);
    }
}

package com.sc.scroe.masteraccount.specification;

import com.sc.scroe.masteraccount.entity.MasterAccountEntity;
import jakarta.persistence.criteria.Predicate;
import org.springframework.data.jpa.domain.Specification;

import java.util.ArrayList;
import java.util.List;

public class MasterAccountSpecification {

    public static Specification<MasterAccountEntity> filterBy(String clientId, String dealId, String masterAccountNumber, String masterAccountCountry, String masterAccountCurrencyCode) {
        return (root, query, cb) -> {
            List<Predicate> predicates = new ArrayList<>();

            // Always fetch only non-deleted records
            predicates.add(cb.isFalse(root.get("deleteFlag")));

            if (clientId != null && !clientId.isBlank()) {
                predicates.add(cb.equal(root.get("clientId"), clientId.trim()));
            }

            if (dealId != null && !dealId.isBlank()) {
                predicates.add(cb.equal(root.get("dealId"), dealId.trim()));
            }

            if (masterAccountNumber != null && !masterAccountNumber.isBlank()) {
                predicates.add(cb.equal(root.get("accountNumber"), masterAccountNumber.trim()));
            }

            if (masterAccountCountry != null && !masterAccountCountry.isBlank()) {
                predicates.add(cb.equal(root.get("countryCode"), masterAccountCountry.trim()));
            }

            if (masterAccountCurrencyCode != null && !masterAccountCurrencyCode.isBlank()) {
                predicates.add(cb.equal(root.get("currencyCode"), masterAccountCurrencyCode.trim()));
            }

            return cb.and(predicates.toArray(new Predicate[0]));
        };
    }
}
