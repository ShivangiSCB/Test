@EnableTransactionManagement
@Repository
@Log4j2
public class ClientParamDAO extends DatabaseAccess {
    private SessionFactory sf;

    public ClientParamDAO(SessionFactory sessionFactory) {
        super(sessionFactory);
        this.sf = sessionFactory;
    }

    @Transactional
    public List<ClientParamTbl> fetchValue(String key) {
        CriteriaBuilder builder = this.sf.getCriteriaBuilder();
        CriteriaQuery<ClientParamTbl> createQuery = builder.createQuery(ClientParamTbl.class);
        Root<ClientParamTbl> root = createQuery.from(ClientParamTbl.class);
        List<Predicate> predicates = new ArrayList<>();
        predicates.add(builder.equal(root.get("clientKeyId").get("key"), key));
        createQuery.select(root).where(predicates.toArray(new Predicate[]{}));
        Query<ClientParamTbl> query = this.sf.getCurrentSession().createQuery(createQuery);
        return query.list();
    }

    @Transactional
    public String fetchValueForClientKey(String key, String clientId) {
        CriteriaBuilder builder = this.sf.getCriteriaBuilder();
        CriteriaQuery<ClientParamTbl> createQuery = builder.createQuery(ClientParamTbl.class);
        Root<ClientParamTbl> root = createQuery.from(ClientParamTbl.class);
        List<Predicate> predicates = new ArrayList<>();
        predicates.add(builder.equal(root.get("clientKeyId").get("key"), key));
        predicates.add(builder.equal(root.get("clientKeyId").get("clientId"), clientId));
        createQuery.select(root).where(predicates.toArray(new Predicate[]{}));
        Query<ClientParamTbl> query = this.sf.getCurrentSession().createQuery(createQuery);
        if (query.list().isEmpty()) {
            log.error("=========ERROR LOG START==========");
            log.error("Error : Value for clientId " + clientId + " and key " + key + " combination is missing in Params Table");
            log.error("=========ERROR LOG END==========");
            return "";
        }
        return query.list().get(0).getValue();
    }

    @Transactional
    public List<ClientParamTbl> getClientParamList(String key, String value) {
        log.info("Begin function getClientParamList key " + key + ",value " + value);
        CriteriaBuilder builder = this.sf.getCriteriaBuilder();
        CriteriaQuery<ClientParamTbl> createQuery = builder.createQuery(ClientParamTbl.class);
        Root<ClientParamTbl> root = createQuery.from(ClientParamTbl.class);
        List<Predicate> predicates = new ArrayList<>();
        predicates.add(builder.equal(root.get("clientKeyId").get("key"), key));
        predicates.add(builder.equal(root.get("value"), value));
        createQuery.select(root).where(predicates.toArray(new Predicate[]{}));
        Query<ClientParamTbl> query = this.sf.getCurrentSession().createQuery(createQuery);

        log.info("Value for key exist in DB " + query.list());
        return query.list();
    }

    @Transactional
    public List<ClientParamTbl> fetchAllConfigForClient(String clientId) {
        CriteriaBuilder builder = this.sf.getCriteriaBuilder();
        CriteriaQuery<ClientParamTbl> createQuery = builder.createQuery(ClientParamTbl.class);
        Root<ClientParamTbl> root = createQuery.from(ClientParamTbl.class);
        List<Predicate> predicates = new ArrayList<>();
        predicates.add(builder.equal(root.get("clientKeyId").get("clientId"), clientId));
        createQuery.select(root).where(predicates.toArray(new Predicate[]{}));
        Query<ClientParamTbl> query = this.sf.getCurrentSession().createQuery(createQuery);
        return query.list();
    }

}

    @Transactional
    public VaMaintainance getVirtualAccountDetails(String virtualAccount) throws Exception {

        log.info(" getVirtualAccountDetails begin " + virtualAccount);
        VaMaintainance vaMaintainance = null;

        DBQuery dbQuery = new DBQuery();
        dbQuery.addFilter(QueryFilter.builder(String.class).key("virtualAccount").value(virtualAccount).operator(FilterOperator.EQ).build());

        List<VaMaintainance> vaMaintainanceList = fetchAllWithCondition(VaMaintainance.class, dbQuery);

        log.info(" vaMaintainanceList from DB " + vaMaintainanceList);

            vaMaintainance = vaMaintainanceList.get(0);
            log.info(" vaMaintainance record from DB " + vaMaintainance);
        }

        return vaMaintainance;

    }

    @Transactional
    public boolean updateVirtualAccountStatus(String virtualAccount, String platformRefId, String status) throws Exception {

        log.info(" updateVirtualAccountStatus begin " + virtualAccount);

        DBQuery dbQuery = new DBQuery();
        dbQuery.addFilter(QueryFilter.builder(String.class).key("virtualAccount").value(virtualAccount).operator(FilterOperator.EQ).build());

        List<VaMaintainance> vaMaintainanceList = fetchAllWithCondition(VaMaintainance.class, dbQuery);
        log.info("vaMaintainanceList from db - " + vaMaintainanceList);
        if (vaMaintainanceList != null && vaMaintainanceList.size() > 0) {
//            String vid = vaMaintainanceList.get(0).getVid();
//            log.info("vaMaintainanceList retire from db vid" + vid);
//            VaMaintainance vaMaintainance = fetchById(VaMaintainance.class, vid).get();
            VaMaintainance vaMaintainance = vaMaintainanceList.get(0);
            log.info("vaMaintainance record from DB - " + vaMaintainance);
            evict(vaMaintainance);
            flush();

            vaMaintainance.setUsedStatus(status);
            vaMaintainance.setPlatFormRef(platformRefId);
            merge(vaMaintainance);
            flush();
            log.info("updateVirtualAccountStatus va maintainance record after update from DB" + vaMaintainance);
        } else {
            log.info("updateVirtualAccountStatus end account does not exist in DB");
            return false;
        }
        return true;

    }

@Component
@Log4j2
public class AdapterCommonUtil {
    public String convertApiErrorToJSONString(List<ApiError> fieldErrorMapList, Class<?> toValueType) {
        log.info("convertApiErrorToJSONString start ApiErrorMapList" + fieldErrorMapList);

        String jsonToString = "Not able to convert " + toValueType + " to json text";

        String jsonArray = "[";
        for (int i = 0; i < fieldErrorMapList.size(); i++) {
            ObjectMapper mapper = new ObjectMapper();
            try {
                mapper.convertValue(fieldErrorMapList.get(i), toValueType);
                jsonToString = mapper.writeValueAsString(mapper.convertValue(fieldErrorMapList.get(i), toValueType));
            } catch (JsonProcessingException e) {
                log.info("Not able to convert json object" + e.getMessage());
                jsonToString = "Not able to convert json object check log";
            }
            jsonArray = jsonArray + jsonToString + ",";
        }
        jsonArray = jsonArray.substring(0, jsonArray.length() - 1) + "]";
        log.info("convertObjectToJSON - " + jsonArray);
        log.info("convertApiErrorToJSONString end ");

        return jsonArray;
    }

        LocalDateTime now = LocalDateTime.now();
        //TODO: System.out.println(dtf.format(now));
        dateTime = dtf.format(now);
        log.info(dateTime);

    }

package com.scb.scroe.adapter.util;

import com.fasterxml.jackson.core.JsonProcessingException;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.databind.SerializationFeature;
import com.jayway.jsonpath.Configuration;
import com.jayway.jsonpath.DocumentContext;
import com.jayway.jsonpath.JsonPath;
import com.jayway.jsonpath.spi.json.JacksonJsonProvider;

public class JsonContext {

    public static DocumentContext createContext(Object data) {

        ObjectMapper mapper = new ObjectMapper();
        mapper.disable(SerializationFeature.FAIL_ON_EMPTY_BEANS);
        Configuration conf = Configuration.builder()
                .jsonProvider(new JacksonJsonProvider())
                .build();
        return JsonPath.using(conf).parse(data.toString());
    }

    public static void write(DocumentContext data, String path, Object value) {
        String[] pathArray = path.split("\\.");
        int length = pathArray.length;
        StringBuilder pathToUpdate = new StringBuilder();
        for (int i = 0; i < length - 1; i++) {
            pathToUpdate.append(pathArray[i]);
            pathToUpdate.append(".");
        }
        data.put(pathToUpdate.toString().substring(0, pathToUpdate.length() - 1), pathArray[length - 1], value);
    }

    public static <T> T createObject(DocumentContext context, Class<T> tClass) throws JsonProcessingException {
        ObjectMapper mapper = new ObjectMapper();
        String jsonContext = context.jsonString();
        return mapper.readValue(jsonContext, tClass);
    }
}

    public AdapterResponse incorrectUrlResponse(ResourceType resourceType) {
        AdapterResponse adapterResponse = new AdapterResponse();
        switch (resourceType) {
            case UNKNOWN_REQUEST_TYPE:
                log.error("==========================  ERROR LOG ============================");
                log.error("Unknown Request type encountered");
                ApiError apiError = new ApiError();
                apiError.setErrorDesc(ErrorCodes.INCORRECT_REQUEST);
                apiError.setErrorCode(ErrorCodes.INCORRECT_REQUEST_CODE);
                List<ApiError> apiErrorList = new ArrayList<>();
                apiErrorList.add(apiError);
                String errorString = commonUtil.convertApiErrorToJSONString(apiErrorList, ApiError.class).replace("\\", "");
                log.error("==========================  ERROR LOG END =========================");
                adapterResponse.setResponse(errorString);
                adapterResponse.setStatus(AdapterConstants.FILE_TRACKING_STATUS_REJECTED);
                return adapterResponse;

    public GetContractResponse getContract(String sql, MultiValueMap<String, String> params) throws JsonProcessingException {
        NativeQuery sqlQuery = sessionFactory.getCurrentSession()
                .createNativeQuery(sql);
        String countQuery = "select count (end_to_end_id) from connect_contract_tbl";
        String whereClause = StringUtils.substringBetween(sql, "where", "order by");
        if (whereClause == null) {
            whereClause = StringUtils.substringAfter(sql, "where");
        }
        if (whereClause != null && !whereClause.isEmpty()) {
            countQuery += (" where " + whereClause);
        }
        NativeQuery countQ = sessionFactory.getCurrentSession()
                .createNativeQuery(countQuery);
        Long totalCount = (Long) (countQ.uniqueResult());
        sqlQuery.addEntity(ContractEntity.class);
        int page = 1, count = 10;
        if (params.containsKey("count"))
            count = Integer.parseInt(params.get("count")
                    .get(0));
        if (params.containsKey("page"))
            page = Integer.parseInt(params.get("page")
                    .get(0));
        sqlQuery.setFirstResult((page - 1) * count);
        sqlQuery.setMaxResults(count);
        List<ContractEntity> contracts = sqlQuery.list();
        ObjectMapper om = new ObjectMapper();
        om.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
        List<GetContractData> contractDataList = new ArrayList<>();
        for (ContractEntity c : contracts) {
            ContractData contractData = om.readValue(c.getContract(), ContractData.class);
            List<EndToEndIdResponse> endToEndIdResponses = new ArrayList<>();
            try {
                List<PaymentResponse> paymentResponse = om.readValue(c.getResponse(), new TypeReference<List<PaymentResponse>>() {
                });
                endToEndIdResponses = paymentResponse.stream()
                        .map(
                                response -> new EndToEndIdResponse(response.getStatus(), response.getEndToEndId(), response.getPlatformRefNo(), response.getFragment()
                                        .stream()
                                        .map(fragment -> new GetContractFragment(fragment.getFragmentPlatformRefNo(), fragment.getStatus(), fragment.getScheduledOn()))
                                        .collect(Collectors.toList())))
                        .collect(Collectors.toList());
            } catch (Exception ignore) {
                log.info("Exception ignored as this contract body although correct, is non-parsable according to current contract payload");
            }
            GetContractData getContractData = new GetContractData(c.getStatus(), contractData, endToEndIdResponses);
            contractDataList.add(getContractData);
        }
        return new GetContractResponse(page, count, totalCount, contractDataList);
    }

        if (clientSetupList.size() != 1) {
            log.error("incorrect setup for scroeClientId -> {}, it has {} rows", vaPartA, clientSetupList.size());
            throw new ValueInDBNotSupportedException(CLIENT_ID_SETUP_INCORRECT);
        }
        return clientSetupList.get(0);
    }

    public String fetchEventCategory(String eventType) {
        DBQuery dbQuery = new DBQuery();
        dbQuery.addFilter(QueryFilter.builder(Object.class)
                .key("eventType")
                .value(eventType)
                .operator(FilterOperator.EQ)
                .build());
        Optional<ConnectEventTypes> connectEventTypes = fetchOneWithCondition(ConnectEventTypes.class, dbQuery);
        if (connectEventTypes.isPresent())
            return connectEventTypes.get()
                    .getEventCategory();
        return eventType;
    }

import java.util.List;
import java.util.Optional;

@Log4j2
@Service
public class XcroWebhookService {
    private final AdapterDAO adapterDAO;
    private final VaultMgmtApi vaultMgmtApi;

    private final ObjectMapper mapper;

    public XcroWebhookService(AdapterDAO adapterDAO, VaultMgmtApi vaultMgmtApi) {
        this.adapterDAO = adapterDAO;
        this.vaultMgmtApi = vaultMgmtApi;
        mapper = new ObjectMapper();
        mapper.disable(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES);
    }

    public Object vaultStatusChange(String webhookString, String clientId) {
        DocumentContext documentContext = JsonContext.createContext(webhookString);
        String status = documentContext.read("status");
        String partyId = documentContext.read("partyId");
        List<PartyVaultMapping> partyVaultMappingList = this.adapterDAO.fetchVaultWithPartyId(partyId);
        for(PartyVaultMapping partyVaultMapping : partyVaultMappingList){
            RefRequest request = new RefRequest();
            request.setClientId(fetchVaultClientId(clientId));
            request.setVaultId(partyVaultMapping.getVaultId());
            request.setRefNo(status);
            VaultWrapperObject vaultWrapperObject = vaultMgmtApi.amendVaultStatus(request);
        }
        return null;
    }

    private String fetchVaultClientId(String clientId) {
        Optional<VaultClientMapping> vaultClientMapping = adapterDAO.fetchClientVaultDetailsById(clientId);
        if (!vaultClientMapping.isPresent()) {
            throw new VaultClientNotFoundException("Vault Client Id not found");
        }
        return vaultClientMapping.get()
                .getVaultClientId();
    }
}
