        response.setPage(pageResult.getNumber() + 1);
        response.setPageCount(pageResult.getTotalPages());
        response.setTotalCount(pageResult.getTotalElements());
        response.setData(pageResult.getContent().stream().map(this::toMasterData).toList());

package com.sc.scroe.masteraccount.specification;

import com.sc.scroe.masteraccount.entity.MasterAccountEntity;
import jakarta.persistence.criteria.Predicate;
import org.springframework.data.jpa.domain.Specification;

import java.util.ArrayList;
import java.util.List;

public class MasterAccountSpecification {

    public static Specification<MasterAccountEntity> filterBy(String clientId, String dealId, String masterAccountNumber, String masterAccountCountry, String masterAccountCurrencyCode) {
        return (root, query, cb) -> {
            List<Predicate> predicates = new ArrayList<>();

            // Always fetch only non-deleted records
            predicates.add(cb.isFalse(root.get("deleteFlag")));

            if (clientId != null && !clientId.isBlank()) {
                predicates.add(cb.equal(root.get("clientId"), clientId.trim()));
            }

            if (dealId != null && !dealId.isBlank()) {
                predicates.add(cb.equal(root.get("dealId"), dealId.trim()));
            }

            if (masterAccountNumber != null && !masterAccountNumber.isBlank()) {
                predicates.add(cb.equal(root.get("accountNumber"), masterAccountNumber.trim()));
            }

            if (masterAccountCountry != null && !masterAccountCountry.isBlank()) {
                predicates.add(cb.equal(root.get("countryCode"), masterAccountCountry.trim()));
            }

            if (masterAccountCurrencyCode != null && !masterAccountCurrencyCode.isBlank()) {
                predicates.add(cb.equal(root.get("currencyCode"), masterAccountCurrencyCode.trim()));
            }

            return cb.and(predicates.toArray(new Predicate[0]));
        };
    }
}
