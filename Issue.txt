package com.sc.scroe.controller;

import com.sc.scroe.masteraccount.dto.*;
import com.sc.scroe.masteraccount.service.MasterAccountService;

import com.sc.scroe.shared.web.NoRecordsResponseUtil;
import jakarta.validation.Valid;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/v1/subledger/masteraccounts")
public class MasterAccountController {

    private static final Logger logger = LoggerFactory.getLogger(MasterAccountController.class);

    @Autowired
    MasterAccountService subledgerMasterAccountService;

    @PostMapping(consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
    public ResponseEntity<CreateMasterAccountResponse> create(@Valid @RequestBody CreateMasterAccountRequest request) {
        CreateMasterAccountResponse masterAccountPostResponse = subledgerMasterAccountService.createMasterAccount(request);
        logger.info("Created Master Account for accountNumber: {}", masterAccountPostResponse.getMasterAccountId());
        return new ResponseEntity<>(new CreateMasterAccountResponse(masterAccountPostResponse.getMasterAccountId()), HttpStatus.CREATED);
    }

    @PutMapping(path = "/{masterAccountId}", consumes = MediaType.APPLICATION_JSON_VALUE, produces = MediaType.APPLICATION_JSON_VALUE)
    public ResponseEntity<UpdateMasterAccountResponse> updateMasterAccount(@PathVariable Long masterAccountId, @Valid @RequestBody UpdateMasterAccountRequest request) {

        UpdateMasterAccountResponse response = subledgerMasterAccountService.updateMasterAccount(masterAccountId, request);

        return ResponseEntity.ok(response);
    }


    @GetMapping(produces = MediaType.APPLICATION_JSON_VALUE)
    public ResponseEntity<?> getMasterAccount(@RequestParam(value = "dealId", required = false) String dealId,
                                              @RequestParam(value = "clientId", required = false) String clientId,
                                              @RequestParam(value = "masterAccountNumber", required = false) String masterAccountNumber,
                                              @RequestParam(value = "masterAccountCountry", required = false) String masterAccountCountry,
                                              @RequestParam(value = "masterAccountCurrencyCode", required = false) String masterAccountCurrencyCode,
                                              @RequestParam(value = "page", defaultValue = "1") Integer page,
                                              @RequestParam(value = "pageSize", defaultValue = "10") Integer pageSize) {
        GetMasterAccountResponse getMasterAccountResponse = subledgerMasterAccountService.getMasterAccount(dealId, clientId, masterAccountNumber, masterAccountCountry, masterAccountCurrencyCode, page, pageSize);
        return NoRecordsResponseUtil.returnResponseOrThrowIfEmpty(getMasterAccountResponse, getMasterAccountResponse.getData(), "No Master account records found", NoRecordsResponseUtil.formatCriterion("dealId", dealId), NoRecordsResponseUtil.formatCriterion("clientId", clientId), NoRecordsResponseUtil.formatCriterion("masterAccountNumber", masterAccountNumber), NoRecordsResponseUtil.formatCriterion("masterAccountCountry", masterAccountCountry), NoRecordsResponseUtil.formatCriterion("masterAccountCurrencyCode", masterAccountCurrencyCode));
    }
}
