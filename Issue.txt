package com.scb.scroeconnect.solaceworker.config;

import com.solacesystems.jms.SolConnectionFactory;
import com.solacesystems.jms.SolJmsUtility;
import jakarta.annotation.PreDestroy;
import jakarta.jms.DeliveryMode;
import lombok.extern.log4j.Log4j2;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.jms.connection.CachingConnectionFactory;
import org.springframework.jms.core.JmsTemplate;

import java.util.Map;

@Log4j2
@Configuration
public class SolaceConnectionConfig {
    private final SolaceProperties properties;
    private final SolaceThreadProperties threadProperties;

    /**
     * Held as a field so @PreDestroy can destroy it on application shutdown,
     * flushing cached sessions and closing the underlying Solace connection cleanly.
     */
    private CachingConnectionFactory cachingConnectionFactory;

    public SolaceConnectionConfig(SolaceProperties properties,
                                  SolaceThreadProperties threadProperties) {
        this.properties = properties;
        this.threadProperties = threadProperties;
    }

    @Bean(name = "solaceConnectionFactory")
    public SolConnectionFactory solaceConnectionFactory() throws Exception {
        Map<String, String> ssl = properties.getSsl();
        Map<String, String> queue = properties.getQueue();

        SolConnectionFactory cf = SolJmsUtility.createConnectionFactory();

        // --- Basic connection ---
        cf.setVPN(queue.get("vpn"));
        cf.setHost(queue.get("hostname")); // supports HA: "host1,host2"
        cf.setUsername(queue.get("username"));
        cf.setPassword(queue.get("password"));
        cf.setClientID(queue.get("clientId"));

        // --- Initial connection retries ---
        // Retries when first establishing a connection on app startup.
        cf.setConnectRetries(properties.getConnectRetries());
        cf.setConnectRetriesPerHost(properties.getConnectRetriesPerHost());
        cf.setConnectTimeoutInMillis(properties.getConnectTimeoutInMillis());

        // --- Reconnect settings (transport/TCP layer) ---
        // Handles reconnection after a LIVE connection drops (broker restart, network blip).
        // 60 retries * 3000ms wait = up to 3 minutes of reconnect attempts before giving up.
        // This is separate from Spring's CachingConnectionFactory.reconnectOnException —
        // Solace handles the physical TCP wire; Spring handles stale cached session eviction.
        cf.setReconnectRetries(properties.getReconnectRetries());
        cf.setReconnectRetryWaitInMillis(properties.getReconnectRetryWaitInMillis());

        // --- Keep-alive / heartbeat ---
        // Detects silently dead connections (e.g. firewall silently drops idle TCP).
        // Solace API only exposes the interval — the limit/threshold is broker-side config.
        cf.setKeepAliveIntervalInMillis(properties.getKeepAliveIntervalInMillis());

        // --- SSL / mTLS ---
        cf.setSSLTrustStore(ssl.get("truststorePath"));
        cf.setSSLTrustStorePassword(ssl.get("truststorePassword"));
        cf.setSSLTrustStoreFormat(ssl.get("truststoreType"));
        cf.setSSLKeyStore(ssl.get("keystorePath"));
        cf.setSSLKeyStorePassword(ssl.get("keystorePassword"));
        cf.setSSLKeyStoreFormat(ssl.get("keystoreType"));
        cf.setAuthenticationScheme("AUTHENTICATION_SCHEME_CLIENT_CERTIFICATE");

        log.info("Solace Connection Factory created");
        return cf;
    }

    @Bean(name = "solaceCachingConnectionFactory")
    public CachingConnectionFactory cachingConnectionFactory(
            @Qualifier("solaceConnectionFactory") SolConnectionFactory cf) {
        CachingConnectionFactory factory = new CachingConnectionFactory(cf);

        // --- Session cache ---
        // Sessions and Producers are NOT shared across threads. CachingConnectionFactory
        // is a borrow/return pool: each thread exclusively borrows one Session for the
        // duration of a send(), then returns it. Two threads never share the same Session.
        // sessionCacheSize = max concurrent sessions = should match executor thread pool size.
        factory.setSessionCacheSize(threadProperties.getCacheSize());

        // Cache MessageProducers per session — avoids recreating them on every send().
        // Same borrow/return guarantee: one producer per session, one session per thread.
        factory.setCacheProducers(true);

        // --- Stale session eviction (Spring cache layer) ---
        // Even after Solace reconnects the TCP wire, old Session objects in the cache
        // are stale (created before the disconnect). Without this, threads keep borrowing
        // broken sessions. With this, a JMSException causes the broken session to be
        // evicted and a fresh one created on the new connection.
        // Works in conjunction with Solace-level reconnect — they solve different layers.
        factory.setReconnectOnException(true);

        this.cachingConnectionFactory = factory;
        return factory;
    }

    @Bean(name = "solaceJmsTemplate")
    public JmsTemplate jmsTemplate(
            @Qualifier("solaceCachingConnectionFactory") CachingConnectionFactory cf) {
        JmsTemplate template = new JmsTemplate(cf);

        // true = publish to Topic destination; false = Queue.
        // Even though durability lives inside broker-side queues that subscribe to topics,
        // the producer MUST address a Topic object (not a Queue object). Without this,
        // JmsTemplate calls session.createQueue() instead of session.createTopic(),
        // and Solace treats it as a direct queue lookup — wrong destination type.
        template.setPubSubDomain(true);

        // PERSISTENT delivery mode: broker writes message to durable store BEFORE
        // routing to subscribed queues and before acking the producer.
        // If broker crashes mid-route, message is recovered from store.
        // NON_PERSISTENT (default) messages are NOT spooled into subscribed queues on Solace —
        // even if the target queues are durable, a non-persistent inbound message is dropped
        // on broker failure before it reaches the queue.
        template.setDeliveryMode(DeliveryMode.PERSISTENT);

        // explicitQosEnabled=true tells JmsTemplate to explicitly pass deliveryMode,
        // priority, and timeToLive into every producer.send(...) call.
        // Without this, setDeliveryMode() above is SILENTLY IGNORED and Solace defaults apply.
        // TTL is not set here (defaults to 0 = never expires) — messages live until consumed.
        // Priority is not set here (defaults to 4 = normal) — all messages equally important.
        template.setExplicitQosEnabled(true);

        return template;
    }

    /**
     * Graceful shutdown on application stop.
     * Destroys CachingConnectionFactory: flushes all cached sessions and closes
     * the underlying Solace connection cleanly, preventing message loss on shutdown.
     */
    @PreDestroy
    public void shutdown() {
        if (cachingConnectionFactory != null) {
            log.info("Shutting down Solace CachingConnectionFactory...");
            cachingConnectionFactory.destroy();
        }
    }
}

import jakarta.persistence.AttributeConverter;
import jakarta.persistence.Converter;
import tools.jackson.databind.JsonNode;
import tools.jackson.databind.ObjectMapper;

@Converter(autoApply = false)
public class JsonNodeConverter implements AttributeConverter<JsonNode, String> {

    private static final ObjectMapper objectMapper = new ObjectMapper();

    @Override
    public String convertToDatabaseColumn(JsonNode attribute) {
        if (attribute == null) return null;
        try {
            return attribute.toString();
        } catch (Exception e) {
            throw new RuntimeException("Failed to serialize JsonNode to text", e);
        }
    }

    @Override
    public JsonNode convertToEntityAttribute(String dbData) {
        if (dbData == null) return null;
        try {
            return objectMapper.readTree(dbData);
        } catch (Exception e) {
            throw new RuntimeException("Failed to deserialize JSON from DB column", e);
        }
    }
}

package com.scb.scroeconnect.solaceworker.mapper;

import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.*;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedWebhookBodyDto;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedWebhookPayloadDto;
import com.scb.scroeconnect.solaceworker.entity.ConnectNotificationDelivery;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;
import tools.jackson.databind.JsonNode;
import tools.jackson.databind.ObjectMapper;

import java.util.UUID;

@Component
@RequiredArgsConstructor
public class FinalWebhookPayloadMapper {

    private final ObjectMapper objectMapper;

    public FinalWebhookPayloadDto buildFinalPayload(TransformedWebhookPayloadDto payload, ConnectNotificationDelivery delivery) {
        FinalWebhookPayloadDto finalPayload = new FinalWebhookPayloadDto();
        FinalWebhookHeaderDto header = buildHeader(payload, delivery);
        FinalWebhookBodyDto body = buildBody(payload);
        finalPayload.setHeader(header);
        finalPayload.setData(body);
        return finalPayload;
    }

    private FinalWebhookHeaderDto buildHeader(TransformedWebhookPayloadDto payload, ConnectNotificationDelivery delivery) {
        FinalWebhookHeaderDto header = new FinalWebhookHeaderDto();
        String msgId = String.valueOf(UUID.randomUUID());
        header.setScroeClientId(payload.getHeader().getScroeClientId());
        header.setWebhookType(payload.getHeader().getWebhookType());
        header.setClientGroupId(payload.getHeader().getClientGroupId());
        header.setMessageId(msgId);
        return header;
    }

    private FinalWebhookBodyDto buildBody(TransformedWebhookPayloadDto payload) {
        TransformedWebhookBodyDto payloadBody = payload.getData();
        FinalWebhookBodyDto body = new FinalWebhookBodyDto();
        body.setDealRefId(payloadBody.getDealRefId());
        body.setEventType(payloadBody.getEventType());
        body.setDealCountry(payloadBody.getDealCountry());
        try {
            JsonNode eventDetailsNode = objectMapper.readTree(payloadBody.getEventDetails());

            body.setEventDetails(eventDetailsNode);

        } catch (Exception e) {
            throw new RuntimeException("Failed to parse eventDetails", e);
        }

        return body;
    }
}

import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.FinalWebhookPayloadDto;

import lombok.AllArgsConstructor;

import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.FinalPayloadDto;

import com.scb.scroeconnect.solaceworker.entity.ConnectNotificationDelivery;
import com.scb.scroeconnect.solaceworker.repository.ConnectNotificationDeliveryRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import tools.jackson.databind.JsonNode;

import java.util.Optional;
import java.util.UUID;

@Log4j2
@Service
@AllArgsConstructor
public class SolaceWorkerService {

    private final ConnectNotificationDeliveryRepository connectNotificationDeliveryRepository;
    @Autowired
    private final RetryHandler retryHandler;
    private final PublishToSolaceUtil publishToSolace;
    private final ParseTransformedPayloadUtil parseTransformedPayload;
    private final SaveFinalPayloadUtil saveFinalPayloadUtil;
    private final FinalPayloadMapper finalPayloadMapper;
    private final ConnectSolaceConfigRepository connectSolaceConfigRepository;
    private final FinalWebhookPayloadMapper finalWebhookPayloadMapper;
    private final PublishToWebhookUtil publishToWebhook;

    @Value("${solace.s2bl.recipientId}")
    private final int solaceRecipientId;

    @Value("${solace.webhook.recipientId}")
    private final int webhookRecipientId;

    @Transactional
    public void sendSolaceNotification(ConnectNotificationDelivery delivery) {
        try {
            log.info("Processing Notification for Delivery ID: {}", delivery.getId());

            JsonNode transformedPayload = delivery.getPayload();

            Long recipientId = delivery.getRecipient().getId();
            UUID eventId = delivery.getNotificationEvent().getEventId();

            Optional<ConnectSolaceConfig> configOpt = connectSolaceConfigRepository.findByRecipientIdAndActive(recipientId);

            if (configOpt.isEmpty()) {
                log.error("No topic config found for recipientId: {}", recipientId);
                retryHandler.handleFailure(delivery);
                return;
            }

            String topic = configOpt.get().getTopic();

            if (recipientId == solaceRecipientId) {
                // SOLACE FLOW
                TransformedPayloadDto payload = parseTransformedPayload.parseTransformedPayload(transformedPayload);

                FinalPayloadDto finalPayload = finalPayloadMapper.buildFinalPayload(payload, delivery);

                boolean published = publishToSolace.publishToSolace(finalPayload, eventId, topic);

                handleResult(delivery, published, finalPayload, "Solace");

            } else if (recipientId == webhookRecipientId) {
                // WEBHOOK FLOW
                TransformedWebhookPayloadDto payload = parseTransformedPayload.parseWebhookPayload(transformedPayload);

                FinalWebhookPayloadDto finalPayload = finalWebhookPayloadMapper.buildFinalPayload(payload, delivery);

                boolean published = publishToWebhook.publishToSolace(finalPayload, eventId, topic);

                handleResult(delivery, published, finalPayload, "Webhook");


            } else {
                log.error("Failed to send Notification for eventId: {}", eventId);
                retryHandler.handleFailure(delivery);
            }

        } catch (Exception e) {
            log.error("ALERT TO PSS:- Failed to process notification: {}", e.getMessage(), e);
            throw new RuntimeException("Failed to send Notification", e);
        }
    }

    private void handleResult(ConnectNotificationDelivery delivery, boolean published, Object payload, String type) {
        if (published) {
            log.info("{} Notification sent successfully for delivery ID: {}", type, delivery.getId());

            connectNotificationDeliveryRepository.updateStatusById(delivery.getId(), DeliveryStatus.SENT.toString());

            saveFinalPayloadUtil.saveFinalPayload(delivery, payload, type);

        } else {
            log.error("Failed to send {} Notification delivery ID: {}", type, delivery.getId());
            retryHandler.handleFailure(delivery);
        }
    }

}

    public boolean publishToSolace(Object finalPayload, UUID eventId, String topic) {
        try {
            String message = objectMapper.writeValueAsString(finalPayload);

            log.error("Exception occurred while sending Solace Notification: ", e);
            return false;
        }
    }
}
