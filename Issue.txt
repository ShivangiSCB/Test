package com.sc.scroe.masteraccount.entity;

import com.sc.scroe.shared.entity.SharedEntity;
import jakarta.persistence.*;

import java.time.LocalDate;

import lombok.Data;

@Data
@Entity
@Table(name = "subledger_masteraccount")
public class SubledgerMasteraccountEntity extends SharedEntity {

    @Id
    @GeneratedValue(strategy = GenerationType.SEQUENCE, generator = "subledger_masteraccount_seq")
    @SequenceGenerator(name = "subledger_masteraccount_seq", sequenceName = "subledger_masteraccount_seq", allocationSize = 1)
    @Column(name = "masteraccount_id", nullable = false)
    private Long masteraccountId;

    @Column(name = "account_id", nullable = false, length = 50)
    private String accountId;

    @Column(name = "master_number", nullable = false, length = 30)
    private String masterNumber;

    @Column(name = "client_id", nullable = false, length = 100)
    private String clientId;

    @Column(name = "deal_id", nullable = false, length = 30)
    private String dealId;

    @Column(name = "account_number", nullable = false, length = 30)
    private String accountNumber;

    @Column(name = "country_code", nullable = false, length = 2)
    private String countryCode;

    @Column(name = "currency_code", nullable = false, length = 3)
    private String currencyCode;

    @Column(name = "account_branch", length = 20)
    private String accountBranch;

    @Column(name = "company_initial", length = 5)
    private String companyInitial;

    @Column(name = "account_name", length = 110)
    private String accountName;

    @Column(name = "iban_number", length = 34)
    private String ibanNumber;

    @Column(name = "account_status", length = 1)
    private String accountStatus;

    @Column(name = "account_opened_date")
    private LocalDate accountOpenedDate;

    @Column(name = "account_closed_date")
    private LocalDate accountClosedDate;

    @Column(name = "client_type", nullable = false, length = 10)
    private String clientType;

    @Column(name = "account_type", nullable = false, length = 20)
    private String accountType;

    @Column(name = "product_code", length = 20)
    private String productCode;

    @Column(name = "sub_product_code", length = 50)
    private String subProductCode;

    @Column(name = "max_sub_account_limit")
    private Integer maxSubAccountLimit;

    @Column(name = "classification_code", length = 20)
    private String classificationCode;

    @Column(name = "sublite_kyc_cdd_flag", nullable = false, length = 1)
    private String subliteKycCddFlag;

    @Column(name = "sublite_kyc_cdd_risk_code", length = 255)
    private String subliteKycCddRiskCode;

    @Column(name = "sub_default_risk_code_flag", nullable = false, length = 1)
    private String subDefaultRiskCodeFlag;

    @Column(name = "sub_default_risk_code", length = 255)
    private String subDefaultRiskCode;

    @Column(name = "address_inherit_flag", nullable = false, length = 1)
    private String addressInheritFlag;

    @Column(name = "clients_money", nullable = false, length = 1)
    private String clientsMoney;

    @Column(name = "sub_account_closure_offset", nullable = false, length = 1)
    private String subAccountClosureOffset;

    @Column(name = "info_code", nullable = false, length = 255)
    private String infoCode;

    @Column(name = "info_value", length = 255)
    private String infoValue;

    @Column(name = "glid", length = 30)
    private String glid;

    @Column(name = "consolidated_statement_flag", length = 1)
    private String consolidatedStatementFlag;

    @Column(name = "account_category", length = 20)
    private String accountCategory;

    @Column(name = "dsr_closing_id", length = 20)
    private String dsrClosingId;

    @Column(name = "advise_interest_indicator", length = 1)
    private String adviseInterestIndicator;

    @Column(name = "credit_sub_acct_int_to_mast", nullable = false, length = 1)
    private String creditSubAcctIntToMast;

    @Column(name = "credit_interest_code", length = 20)
    private String creditInterestCode;

    @Column(name = "debit_interest_code", length = 20)
    private String debitInterestCode;

    @Column(name = "suppress_statement_flag", nullable = false, length = 1)
    private String suppressStatementFlag;

    @Column(name = "interest_code", nullable = false, length = 20)
    private String interestCode;

    @Column(name = "credit_interest_account_number", length = 40)
    private String creditInterestAccountNumber;

    @Column(name = "credit_interest_currency_code", length = 3)
    private String creditInterestCurrencyCode;
}
