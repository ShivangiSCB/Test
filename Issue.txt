package com.sc.scroe.controller.advice;

import com.sc.scroe.exception.ErrorResponse;
import com.sc.scroe.exception.ResourceNotFoundException;
import com.sc.scroe.shared.web.ApiWebConstants;
import java.util.Arrays;
import java.util.LinkedHashMap;
import java.util.Map;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.dao.DataIntegrityViolationException;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

@RestControllerAdvice
public class GlobalExceptionHandler {

    private static final Logger logger = LoggerFactory.getLogger(GlobalExceptionHandler.class);

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<ErrorResponse> handleValidation(MethodArgumentNotValidException ex) {
        String details = ex.getBindingResult()
                .getFieldErrors()
                .stream()
                .map(err -> err.getField() + ": " + err.getDefaultMessage())
                .reduce((a, b) -> a + "; " + b)
                .orElse(ex.getMessage());
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        Map<String, String> error = new LinkedHashMap<>();
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", details);
        return ResponseEntity
                .status(HttpStatus.BAD_REQUEST)
                .body(new ErrorResponse(Arrays.asList(error)));
    }

    @ExceptionHandler(IllegalArgumentException.class)
    public ResponseEntity<ErrorResponse> handleBadRequest(IllegalArgumentException ex) {
        Map<String, String> error = new LinkedHashMap<>();
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", ex.getMessage());
        return ResponseEntity
                .status(HttpStatus.BAD_REQUEST)
                .body(new ErrorResponse(Arrays.asList(error)));
    }

    @ExceptionHandler(ResourceNotFoundException.class)
    public ResponseEntity<ErrorResponse> handleResourceNotFound(ResourceNotFoundException ex) {
        Map<String, String> error = new LinkedHashMap<>();
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", ex.getMessage());
        return ResponseEntity
                .status(HttpStatus.NOT_FOUND)
                .body(new ErrorResponse(Arrays.asList(error)));
    }

    @ExceptionHandler(DataIntegrityViolationException.class)
    public ResponseEntity<ErrorResponse> handleResourceNotFound(DataIntegrityViolationException ex) {
        Map<String, String> error = new LinkedHashMap<>();
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", ex.getMessage());
        return ResponseEntity
                .status(HttpStatus.NOT_FOUND)
                .body(new ErrorResponse(Arrays.asList(error)));
    }

    @ExceptionHandler(Exception.class)
    public ResponseEntity<ErrorResponse> handleUnexpected(Exception ex) {
        Map<String, String> error = new LinkedHashMap<>();
        logger.error("Unexpected error occurred: {}", ex.getMessage(), ex);
        error.put("code", ApiWebConstants.API_COMMON_ERROR_CODE);
        error.put("description", "Technical Issue - Please contact the support team");
        return ResponseEntity
                .status(HttpStatus.INTERNAL_SERVER_ERROR)
                .body(new ErrorResponse(Arrays.asList(error)));
    }
}
