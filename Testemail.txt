package com.scb.scroeconnect.solaceworker.mapper;

import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.*;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedWebhookBodyDto;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedWebhookPayloadDto;
import com.scb.scroeconnect.solaceworker.entity.ConnectNotificationDelivery;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.UUID;

@Component
public class FinalWebhookPayloadMapper {

    public FinalWebhookPayloadDto buildFinalPayload(TransformedWebhookPayloadDto payload, ConnectNotificationDelivery delivery) {
        FinalWebhookPayloadDto finalPayload = new FinalWebhookPayloadDto();
        FinalWebhookHeaderDto header = buildHeader(payload, delivery);
        FinalWebhookBodyDto body = buildBody(payload);
        finalPayload.setHeader(header);
        finalPayload.setData(List.of(body));
        return finalPayload;
    }

    private FinalWebhookHeaderDto buildHeader(TransformedWebhookPayloadDto payload, ConnectNotificationDelivery delivery) {
        FinalWebhookHeaderDto header = new FinalWebhookHeaderDto();
        String msgId = String.valueOf(UUID.randomUUID());
        header.setScroeClientId(payload.getHeader().getScroeClientId());
        header.setWebhookType(payload.getHeader().getWebhookType());
        header.setClientGroupId(payload.getHeader().getClientGroupId());
        header.setMessageId(msgId);
        return header;
    }

    private FinalWebhookBodyDto buildBody(TransformedWebhookPayloadDto payload) {
        TransformedWebhookBodyDto payloadBody = payload.getBody().get(0);
        FinalWebhookBodyDto body = new FinalWebhookBodyDto();
        String eventDetails = payloadBody.getEventDetails();
        body.setDealRefId(payloadBody.getDealRefId());
        body.setEventType(payloadBody.getEventType());
        body.setDealCountry(payloadBody.getDealCountry());
        body.setEventDetails();
        return body;
    }
}

package com.scb.scroeconnect.solaceworker.service;

import com.scb.scroeconnect.solaceworker.entity.ConnectSolaceConfig;
import com.scb.scroeconnect.solaceworker.enums.DeliveryStatus;
import com.scb.scroeconnect.solaceworker.mapper.FinalPayloadMapper;
import com.scb.scroeconnect.solaceworker.repository.ConnectSolaceConfigRepository;
import com.scb.scroeconnect.solaceworker.util.ParseTransformedPayloadUtil;
import com.scb.scroeconnect.solaceworker.util.PublishToSolaceUtil;
import com.scb.scroeconnect.solaceworker.util.RetryHandler;
import com.scb.scroeconnect.solaceworker.util.SaveFinalPayloadUtil;
import lombok.AllArgsConstructor;
import lombok.extern.log4j.Log4j2;
import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.FinalPayloadDto;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedPayloadDto;
import com.scb.scroeconnect.solaceworker.entity.ConnectNotificationDelivery;
import com.scb.scroeconnect.solaceworker.repository.ConnectNotificationDeliveryRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import tools.jackson.databind.JsonNode;

import java.util.Optional;
import java.util.UUID;

@Log4j2
@Service
@AllArgsConstructor
public class SolaceWorkerService {

    private final ConnectNotificationDeliveryRepository connectNotificationDeliveryRepository;
    @Autowired
    private final RetryHandler retryHandler;
    private final PublishToSolaceUtil publishToSolace;
    private final ParseTransformedPayloadUtil parseTransformedPayload;
    private final SaveFinalPayloadUtil saveFinalPayloadUtil;
    private final FinalPayloadMapper finalPayloadMapper;
    private final ConnectSolaceConfigRepository connectSolaceConfigRepository;

    @Transactional
    public void sendSolaceNotification(ConnectNotificationDelivery connectNotificationDelivery) {
        try {
            log.info("Processing Solace Notification for Delivery ID: {}", connectNotificationDelivery.getId());

            JsonNode transformedPayload = connectNotificationDelivery.getPayload();

            TransformedPayloadDto payload = parseTransformedPayload.parseTransformedPayload(transformedPayload);

            FinalPayloadDto finalPayload = finalPayloadMapper.buildFinalPayload(payload, connectNotificationDelivery);

            Long recipientId = connectNotificationDelivery.getRecipient().getId();
            UUID eventId = connectNotificationDelivery.getNotificationEvent().getEventId();

            Optional<ConnectSolaceConfig> getTopicDetails = connectSolaceConfigRepository.findByRecipientIdAndActive(recipientId);

            if (getTopicDetails.isPresent()) {

                ConnectSolaceConfig connectSolaceConfig = getTopicDetails.get();
                String topic = connectSolaceConfig.getTopic();

                if (recipientId == 1) {

                    boolean solaceNotification = publishToSolace.publishToSolace(finalPayload, recipientId, eventId, topic);

                    if (solaceNotification) {
                        log.info("Solace Notification sent successfully for delivery ID: {}", connectNotificationDelivery.getId());

                        connectNotificationDeliveryRepository.updateStatusById(connectNotificationDelivery.getId(), DeliveryStatus.SENT.toString());

                        saveFinalPayloadUtil.saveFinalPayload(connectNotificationDelivery, finalPayload, "Solace");

                    } else {
                        log.error("Failed to send Solace Notification delivery ID: {}, Implementing Retry Block", connectNotificationDelivery.getId());

                        retryHandler.handleFailure(connectNotificationDelivery);
                    }
                } else if (recipientId == 2) {

                    boolean webhookNotification = publishToSolace.publishToSolace(finalPayload, recipientId, eventId, topic);

                    if (webhookNotification) {
                        log.info("Webhook Notification sent successfully for delivery ID: {}", connectNotificationDelivery.getId());

                        connectNotificationDeliveryRepository.updateStatusById(connectNotificationDelivery.getId(), DeliveryStatus.SENT.toString());

                        saveFinalPayloadUtil.saveFinalPayload(connectNotificationDelivery, finalPayload, "Webhook");

                    } else {
                        log.error("Failed to send Webhook Notification delivery ID: {}, Implementing Retry Block", connectNotificationDelivery.getId());

                        retryHandler.handleFailure(connectNotificationDelivery);
                    }

                } else {
                    log.error("Failed to send Notification for delivery ID: {}, Implementing Retry Block", connectNotificationDelivery.getId());
                }
            }

        } catch (Exception e) {
            log.error("ALERT TO PSS:- Failed to send Solace Notification: {}", e.getMessage(), e);
            throw new RuntimeException("Failed to send Solace Notification", e);
        }
    }

}
