package com.scb.liquidity.messaging.producers;
 
import com.scb.liquidity.entities.MomProperties;
import com.scb.liquidity.entities.SolaceProperties;
import com.scb.liquidity.exceptions.LiquidityMessagingException;
import com.scb.liquidity.messaging.Constants.JMSHeader;
import com.solacesystems.jms.SolConnectionFactory;
import com.solacesystems.jms.message.SolMessage;
import lombok.Getter;
import lombok.Setter;
import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;
 
import javax.jms.*;
import java.util.UUID;
 
@Getter
@Setter
public class SolJmsSolaceProducer implements SolaceProducer {
    private static final Logger LOGGER = LogManager.getLogger(SolJmsSolaceProducer.class);
    private static final Boolean IS_DMQ_ENABLED = false;
    private static final Integer MAX_REDELIVERY_COUNT = 20;
 
    private MomProperties properties;
    private SolConnectionFactory connectionFactory;
    private static ThreadLocal<Session> jmsSession = new ThreadLocal<>();
 
    public SolJmsSolaceProducer(SolaceProperties properties, SolConnectionFactory connectionFactory) {
        this.connectionFactory = connectionFactory;
        this.properties = properties;
    }
 
    public static ThreadLocal<Session> getJmsSession() {
        return jmsSession;
    }
 
    public static void setJmsSession(ThreadLocal<Session> jmsSession) {
        SolJmsSolaceProducer.jmsSession = jmsSession;
    }
 
    public <T> void send(String topicName, T payload) {
        send(topicName, payload, null, null);
    }
 
    public <T> void send(String topicName, T payload, CompletionListener completionListener) {
        send(topicName, payload, null, completionListener);
    }
 
    public <T> void send(String topicName, T payload, JMSHeader header, CompletionListener completionListener) {
        try {
            checkProducerExistence();
            Session session = jmsSession.get();
            Destination destination = session.createTopic(topicName);
            MessageProducer producer = session.createProducer(destination);
            if (payload instanceof String) {
                SolMessage textMessage = (SolMessage) jmsSession.get().createTextMessage((String) payload);
                // START: Mandated by Solace Governance - SCB
                String mId = UUID.randomUUID().toString();
                textMessage.setJMSCorrelationID(mId);
                textMessage.setStringProperty("topic", topicName);
                if (header != null && header.isDMQEnabled())
                    textMessage.setIsDMQEligible(header.isDMQEnabled());
                else
                    textMessage.setIsDMQEligible(IS_DMQ_ENABLED);
                if (header != null && header.isRedeliverable())
                    textMessage.setJMSXDeliveryCount(MAX_REDELIVERY_COUNT);
                if (header != null && header.getConversationId() != null)
                    textMessage.setStringProperty("conversationId", header.getConversationId());
                if (header != null && header.getJMSMessageType() != null) {
                    textMessage.setJMSType(header.getJMSMessageType());
                } else {
                    textMessage.setJMSType("string");
                }
                // END: Mandated by Solace Governance - SCB
                if (header != null) {
                    if (header.getJMSReplyTo() != null)
                        textMessage.setJMSReplyTo(jmsSession.get().createTopic(header.getJMSReplyTo()));
                    if (header.getJMSExpiration() != 0)
                        textMessage.setJMSExpiration(header.getJMSExpiration());
                    if (header.getJMSPriority() != 0)
                        textMessage.setJMSPriority(header.getJMSPriority());
                    if (!header.getJMSString().isEmpty())
                        header.getJMSString().forEach((k, v) -> {
                            try {
                                textMessage.setStringProperty(k, v);
                            } catch (JMSException e) {
                                LOGGER.error("Error in adding property {}:{}", k, v);
                            }
                        });
                    if (!header.getJMSBoolean().isEmpty())
                        header.getJMSBoolean().forEach((k, v) -> {
                            try {
                                textMessage.setBooleanProperty(k, v);
                            } catch (JMSException e) {
                                LOGGER.error("Error in adding property {}:{}", k, v);
                            }
                        });
                    if (!header.getJMSDouble().isEmpty())
                        header.getJMSDouble().forEach((k, v) -> {
                            try {
                                textMessage.setDoubleProperty(k, v);
                            } catch (JMSException e) {
                                LOGGER.error("Error in adding property {}:{}", k, v);
                            }
                        });
                    if (!header.getJMSInt().isEmpty())
                        header.getJMSInt().forEach((k, v) -> {
                            try {
                                textMessage.setIntProperty(k, v);
                            } catch (JMSException e) {
                                LOGGER.error("Error in adding property {}:{}", k, v);
                            }
                        });
                    if (!header.getJMSLong().isEmpty())
                        header.getJMSLong().forEach((k, v) -> {
                            try {
                                textMessage.setLongProperty(k, v);
                            } catch (JMSException e) {
                                LOGGER.error("Error in added property {}:{}", k, v);
                            }
                        });
                }
                textMessage.setJMSTimestamp(System.currentTimeMillis());
                textMessage.setJMSDestination(destination);
                if (completionListener != null)
                    producer.send(textMessage, completionListener);
                else
                    producer.send(textMessage);
                LOGGER.info("Message sent with id: {}, correlationId: {}, timestamp: {}", textMessage.getJMSMessageID(),
                        textMessage.getJMSCorrelationID(), textMessage.getJMSTimestamp());
                LOGGER.debug("Message sent with id: {} -- {}", textMessage.getJMSMessageID(),
                        ((TextMessage) textMessage).getText());
                return;
            }
        } catch (JMSException e) {
            throw new LiquidityMessagingException("Error while sending message", e);
        }
        throw new LiquidityMessagingException("Message Type unsupported");
    }
 
    private void checkProducerExistence() {
        try {
            if (jmsSession.get() == null) {
                Connection connection = this.connectionFactory.createConnection();
                Session session = connection.createSession(false, Session.CLIENT_ACKNOWLEDGE);
                jmsSession.set(session);
            }
        } catch (JMSException e) {
            throw new LiquidityMessagingException("Error in connection instantiation", e);
        }
    }
 
    public void triggerShutdown() {
        try {
            LOGGER.info("Shutdown called for host: {}", properties.getHostname());
            if (jmsSession.get() != null) {
                jmsSession.get().close();
                jmsSession.remove();
            }
            LOGGER.info("Producer for host: {} is shutdown", properties.getHostname());
        } catch (JMSException e) {
            throw new LiquidityMessagingException("Error in closing the connection", e);
        }
    }
}
