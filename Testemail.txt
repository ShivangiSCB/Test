package com.scb.scroeconnect.solaceworker.converter;

import jakarta.persistence.AttributeConverter;
import jakarta.persistence.Converter;
import lombok.AllArgsConstructor;
import org.postgresql.util.PGobject;
import org.springframework.beans.factory.annotation.Autowired;
import tools.jackson.databind.JsonNode;
import tools.jackson.databind.ObjectMapper;

@AllArgsConstructor
@Converter
public class JsonNodeConverter implements AttributeConverter<JsonNode, Object> {

    @Autowired
    private final ObjectMapper objectMapper;

    @Override
    public Object convertToDatabaseColumn(JsonNode attribute) {
        if (attribute == null) return null;
        try {
            PGobject pgObject = new PGobject();
            pgObject.setType("jsonb");               // ← tells JDBC: send as jsonb
            pgObject.setValue(attribute.toString()); // ← serialize JsonNode to JSON string
            return pgObject;
        } catch (Exception e) {
            throw new RuntimeException("Failed to serialize JsonNode to jsonb", e);
        }
    }

    @Override
    public JsonNode convertToEntityAttribute(Object dbData) {
        if (dbData == null) return null;
        try {
            String json = (dbData instanceof PGobject)
                    ? ((PGobject) dbData).getValue()
                    : dbData.toString();
            return objectMapper.readTree(json);
        } catch (Exception e) {
            throw new RuntimeException("Failed to deserialize JSON from DB column", e);
        }
    }
}

package com.scb.scroeconnect.solaceworker.service;

import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.FinalWebhookPayloadDto;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedWebhookPayloadDto;

import lombok.AllArgsConstructor;
import lombok.extern.log4j.Log4j2;
import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.FinalPayloadDto;
import com.scb.scroeconnect.solaceworker.dto.TransformedPayloadDto.TransformedPayloadDto;
import com.scb.scroeconnect.solaceworker.entity.ConnectNotificationDelivery;
import com.scb.scroeconnect.solaceworker.repository.ConnectNotificationDeliveryRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import tools.jackson.databind.JsonNode;

import java.util.Optional;
import java.util.UUID;

@Log4j2
@Service
@AllArgsConstructor
public class SolaceWorkerService {

    private final ConnectNotificationDeliveryRepository connectNotificationDeliveryRepository;
    @Autowired
    private final RetryHandler retryHandler;
    private final PublishToSolaceUtil publishToSolace;
    private final ParseTransformedPayloadUtil parseTransformedPayload;
    private final SaveFinalPayloadUtil saveFinalPayloadUtil;
    private final FinalPayloadMapper finalPayloadMapper;
    private final ConnectSolaceConfigRepository connectSolaceConfigRepository;
    private final FinalWebhookPayloadMapper finalWebhookPayloadMapper;

    @Transactional
    public void sendSolaceNotification(ConnectNotificationDelivery delivery) {

        try {
            log.info("Processing Notification for Delivery ID: {}", delivery.getId());

            JsonNode transformedPayload = delivery.getPayload();

            Long recipientId = delivery.getRecipient().getId();
            UUID eventId = delivery.getNotificationEvent().getEventId();

            Optional<ConnectSolaceConfig> configOpt =
                    connectSolaceConfigRepository.findByRecipientIdAndActive(recipientId);

            if (configOpt.isEmpty()) {
                log.error("No topic config found for recipientId: {}", recipientId);
                retryHandler.handleFailure(delivery);
                return;
            }

            String topic = configOpt.get().getTopic();

            if (recipientId == 1) {
                // SOLACE FLOW
                TransformedPayloadDto payload =
                        parseTransformedPayload.parseTransformedPayload(transformedPayload);

                FinalPayloadDto finalPayload =
                        finalPayloadMapper.buildFinalPayload(payload, delivery);

                boolean published =
                        publishToSolace.publishToSolace(finalPayload, recipientId, eventId, topic);

                handleResult(delivery, published, finalPayload, "Solace");

            } else if (recipientId == 2) {
                // WEBHOOK FLOW
                TransformedWebhookPayloadDto payload =
                        parseTransformedPayload.parseWebhookPayload(transformedPayload);

                FinalWebhookPayloadDto finalPayload =
                        finalWebhookPayloadMapper.buildFinalPayload(payload, delivery);

                boolean published =
                        publishToSolace.publishToSolace(finalPayload, recipientId, eventId, topic);

                handleResult(delivery, published, finalPayload, "Webhook");

            } else {
                log.error("Failed to send Notification for eventId: {}", eventId);
                retryHandler.handleFailure(delivery);
            }

        } catch (Exception e) {
            log.error("ALERT TO PSS:- Failed to process notification: {}", e.getMessage(), e);
            throw new RuntimeException("Failed to send Notification", e);
        }
    }

    private void handleResult(ConnectNotificationDelivery delivery,
                              boolean published,
                              Object payload,
                              String type) {

        if (published) {
            log.info("{} Notification sent successfully for delivery ID: {}", type, delivery.getId());

            connectNotificationDeliveryRepository.updateStatusById(
                    delivery.getId(),
                    DeliveryStatus.SENT.toString()
            );

            saveFinalPayloadUtil.saveFinalPayload(delivery, payload, type);

        } else {
            log.error("Failed to send {} Notification delivery ID: {}", type, delivery.getId());
            retryHandler.handleFailure(delivery);
        }
    }

}

package com.scb.scroeconnect.solaceworker.util;

import com.scb.scroeconnect.solaceworker.dto.FinalPayloadDto.FinalPayloadDto;
import com.scb.scroeconnect.solaceworker.entity.ConnectNotificationDelivery;
import com.scb.scroeconnect.solaceworker.entity.ConnectSolaceWorker;
import com.scb.scroeconnect.solaceworker.entity.NotificationEvent;
import com.scb.scroeconnect.solaceworker.repository.ConnectNotificationEventRepository;
import com.scb.scroeconnect.solaceworker.repository.ConnectSolaceWorkerRepository;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;
import tools.jackson.databind.JsonNode;

import java.time.ZonedDateTime;

@Component
public class SaveFinalPayloadUtil
{
        @Autowired
        ConnectNotificationEventRepository connectNotificationEventRepository;

        @Autowired
        ConnectSolaceWorkerRepository connectSolaceWorkerRepository;

    public void saveFinalPayload(ConnectNotificationDelivery delivery, Object finalPayload, String deliveryMode) throws RuntimeException {
        NotificationEvent notificationEvent = connectNotificationEventRepository
                .findByEventId(delivery.getNotificationEvent().getEventId())
                .orElseThrow(() -> new RuntimeException("ALERT TO PSS, Event not found for ID: " + delivery.getNotificationEvent().getEventId()));

        // Create and populate the ConnectNotificationEmailInfo entity
        ConnectSolaceWorker connectSolaceWorker = new ConnectSolaceWorker();
        connectSolaceWorker.setDeliveryMode(deliveryMode);
        connectSolaceWorker.setNotificationEvent(notificationEvent);
        connectSolaceWorker.setFinalPayload((JsonNode) finalPayload);
        connectSolaceWorker.setSendAt(ZonedDateTime.now());
        connectSolaceWorker.setNotificationDelivery(delivery);

        // Save the entity
        connectSolaceWorkerRepository.save(connectSolaceWorker);
    }

}
